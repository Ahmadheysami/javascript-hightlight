const {Router} = require("express")

const router = new Router()
const authController = require("../controllers/auth")
const accessApi = require("../middlewares/accessApi")
const authenticated = require("../middlewares/authenticated")


/*
* @desc Login User
* @method POST
* @route /api/auth/login
* @body []
* @headers [Accept]
*/
router.post("/auth/login", accessApi, authController.login)


/*
* @desc Login User With Email and password
* @method POST
* @route /api/auth/login/v2
* @body [email , password]
* @headers [ Accept ]
*/
router.post("/auth/login/v2", accessApi, authController.loginV2)


/*
* @desc Verify Otp Password
* @method GET
* @route /api/auth/verify
* @body []
* @headers [ Accept ]
* @query [ otp , token ]
*/
router.get("/auth/verify", accessApi, authController.verify)

/*
* @desc Completion Information User Data
* @method POST
* @route /api/auth/completion-information
* @body [ username ,fullname ,email ]
* @headers [Accept]
*/
router.post("/auth/completion-information" , accessApi, authenticated.user ,  authController.completionInformation)

module.exports = router
